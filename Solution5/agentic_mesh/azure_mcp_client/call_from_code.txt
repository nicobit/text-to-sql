from openai_agents_python import Client, SseTransport

mcp = Client(
    name="my-script",
    version="0.1",
    transport=SseTransport("http://localhost:3000/sse")
)
await mcp.initialize()

# invoke the “azure.resourceGroups.list” tool
result = await mcp.request(
    method="azure.resourceGroups.list",
    params={}
)
print(result)
