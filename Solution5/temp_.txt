3) 

docs/assets/js/mermaid-init.js





Theme variables tuned for “Azure” feel in both schemes. Mermaid v10 supports themeVariables below.

(function () {

  // Read current scheme from Material (updates on toggle)

  function currentScheme() {

    const body = document.body;

    const dark = body.getAttribute('data-md-color-scheme') === 'slate';

    return dark ? 'dark' : 'light';

  }



  function themeVars(scheme) {

    const isDark = scheme === 'dark';



    // Base palette (indigo/azure accents)

    const base = {

      fontFamily: "Inter, system-ui, -apple-system, Segoe UI, Roboto, Arial, sans-serif",

      primaryColor: isDark ? "#4f46e5" : "#4f46e5",

      primaryTextColor: "#ffffff",

      lineColor: isDark ? "rgba(255,255,255,0.5)" : "rgba(0,0,0,0.5)",

      textColor: isDark ? "rgba(255,255,255,0.92)" : "rgba(0,0,0,0.84)",

      background: "transparent",         // card background handled by CSS

      mainBkg: "transparent",

      clusterBkg: "transparent",

      clusterBorder: isDark ? "rgba(255,255,255,0.12)" : "rgba(0,0,0,0.08)",

      borderRadius: 8

    };



    // Gantt specifics

    const gantt = {

      ganttAxisFontSize: "12px",

      ganttTaskHeight: 24,

      ganttBarCornerRadius: 6,

      ganttSectionBkgColor: isDark ? "rgba(255,255,255,0.03)" : "rgba(0,0,0,0.025)",

      ganttSectionBkgColor2: "transparent",

      ganttGridLineColor: isDark ? "rgba(255,255,255,0.12)" : "rgba(0,0,0,0.08)",

      ganttTodayLineColor: isDark ? "#38bdf8" : "#4f46e5",

    };



    // Task states

    const tasks = {

      taskTextColor: base.textColor,

      taskBkgColor: isDark ? "rgba(99,102,241,0.35)" : "rgba(99,102,241,0.18)",   // default

      taskBorderColor: isDark ? "rgba(99,102,241,0.9)" : "rgba(99,102,241,0.65)",

      taskDoneColor: isDark ? "rgba(16,185,129,0.35)" : "rgba(16,185,129,0.18)",

      taskDoneTextColor: base.textColor,

      taskLateColor: isDark ? "rgba(239,68,68,0.35)" : "rgba(239,68,68,0.2)",

      taskLateTextColor: base.textColor,

      taskCriticalColor: isDark ? "rgba(245,158,11,0.35)" : "rgba(245,158,11,0.22)",

      taskCriticalTextColor: base.textColor,

      activeTaskBorderColor: isDark ? "#93c5fd" : "#2563eb"

    };



    return Object.assign({}, base, gantt, tasks);

  }



  function initMermaid() {

    const scheme = currentScheme();

    mermaid.initialize({

      startOnLoad: true,

      theme: "base",               // use 'base' to fully control via themeVariables

      themeVariables: themeVars(scheme),

      gantt: {

        useWidth: 1000,            // allow wide charts; container scrolls

        leftPadding: 85,

        rightPadding: 20,

        topPadding: 30,

        barGap: 6,

        // tickInterval can be '1 week', '1 day', etc.  Keep default auto here.

      },

      securityLevel: "loose"       // allow HTML in labels (safe in docs)

    });

  }



  // Initialize now

  if (window.mermaid) initMermaid();



  // Re-init when Material toggles color scheme

  document.addEventListener("DOMContentLoaded", () => {

    const observer = new MutationObserver((muts) => {

      for (const m of muts) {

        if (m.type === "attributes" && m.attributeName === "data-md-color-scheme") {

          if (window.mermaid) initMermaid();

        }

      }

    });

    observer.observe(document.body, { attributes: true });

  });

})();

# Roadmap (Q4)



<div class="ai-gantt--compact">



```mermaid

gantt

    title Product Milestones (Q4 2025)

    dateFormat  YYYY-MM-DD

    axisFormat  %d %b



    %% Today marker

    todayMarker stroke-width:2,stroke-dasharray: 3 3



    section Core

    Kickoff           :milestone, m1, 2025-10-03, 0d

    Architecture      :active, a1, 2025-10-01, 2025-10-20

    Beta Freeze       :milestone, m2, 2025-11-12, 0d

    GA Release        :crit, milestone, m3, 2025-12-15, 0d



    section Delivery

    Feature A         :done,   f1, 2025-10-05, 2025-10-25

    Feature B         :        f2, 2025-10-15, 2025-11-15

    Hardening         :crit,   h1, 2025-11-16, 2025-12-10

</div>

```

Notes



:done, :crit, :active pick up the green / amber / accent tones you saw in tables.
The container manages the rounded glass card; Mermaid itself is set to theme: base and themed with variables for light/dark.
If your chart is very wide, it will scroll horizontally inside the rounded card.










5) Optional: color “tags” per section





If you want sections tinted differently (e.g., Core vs Delivery), add directives:

gantt

    title Tinted Sections

    dateFormat  YYYY-MM-DD

    %% Section colors (Mermaid v10 supports themeVariables but not per-section vars;

    %% we can emulate with a subtle background task)

    section Core

    %% background strip (0 height visually but paints strip)

    Core BG :done, corebg, 2025-10-01, 1d

    Task 1  : a, 2025-10-01, 10d

    section Delivery

    Delivery BG :done, delbg, 2025-10-01, 1d

    Task 2      : b, 2025-10-15, 8d

(For stronger per-section hues, you can add small CSS overrides targeting gantt .section:nth-of-type(n).)
